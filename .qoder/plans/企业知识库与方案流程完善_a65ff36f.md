## Task 2: 数据库模型扩展 - 企业知识库

**新增 Prisma 模型**（在 `crm-backend/prisma/schema.prisma` 中添加）：

```prisma
// 知识库文档
model KnowledgeDocument {
  id            String   @id @default(cuid())
  title         String
  category      String   // product_pricing | proposal_template | contract_template | custom
  subCategory   String?
  description   String?  @db.Text
  fileUrl       String?
  fileName      String?
  fileType      String?  // xlsx, csv, pdf, docx, png, jpg
  fileSize      Int?
  content       String?  @db.LongText  // 解析后的文本内容
  metadata      Json?    // 额外结构化数据
  tags          Json?    // 标签数组
  isActive      Boolean  @default(true)
  version       Int      @default(1)
  createdById   String
  createdBy     User     @relation("KnowledgeDocCreator", fields: [createdById], references: [id])
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

// 产品价格表
model ProductPricing {
  id            String   @id @default(cuid())
  documentId    String?  // 关联导入文档
  productName   String
  productCode   String?
  category      String?
  specification String?  @db.Text
  unitPrice     Decimal  @db.Decimal(12, 2)
  unit          String?  // 个、套、年
  minQuantity   Int?
  discount      Json?    // 阶梯折扣规则
  validFrom     DateTime?
  validUntil    DateTime?
  isActive      Boolean  @default(true)
  notes         String?  @db.Text
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

// 合同模板
model ContractTemplate {
  id            String   @id @default(cuid())
  name          String
  category      String   // 销售合同、服务协议、保密协议等
  description   String?  @db.Text
  content       String   @db.LongText
  variables     Json?    // 模板变量定义
  tags          Json?
  usageCount    Int      @default(0)
  isActive      Boolean  @default(true)
  createdById   String
  createdBy     User     @relation("ContractTemplateCreator", fields: [createdById], references: [id])
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

// 自定义数据表
model CustomDataTable {
  id            String   @id @default(cuid())
  name          String
  description   String?
  columns       Json     // 列定义: [{name, type, required, options}]
  createdById   String
  createdBy     User     @relation("CustomTableCreator", fields: [createdById], references: [id])
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}

// 自定义数据行
model CustomDataRow {
  id            String   @id @default(cuid())
  tableId       String
  table         CustomDataTable @relation(fields: [tableId], references: [id], onDelete: Cascade)
  data          Json     // 行数据
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
}
```

## 技术决策

| 决策项 | 选择 | 原因 |
|--------|------|------|
| 文件上传 | multer + 本地存储 | 与现有录音上传一致 |
| Excel解析 | xlsx (SheetJS) 库 | 成熟稳定，支持导入导出 |
| CSV解析 | csv-parse 库 | 轻量级 CSV 处理 |
| OCR/文档解析 | 模拟模式优先 | 按用户选择，先实现完整UI和数据流 |
| 知识库搜索 | Prisma全文搜索 | 初期足够，后续可升级为Elasticsearch |
| 前端富文本 | textarea + markdown | 保持轻量，与现有风格一致 |
